macro(test_secant test_arg)
  configure_file(secant${test_arg}.mtest.in
    secant${test_arg}.mtest @ONLY)
  add_test(NAME secant${test_arg}_mtest
    COMMAND mtest --verbose=level0 --xml-output=true --result-file-output=false secant${test_arg}.mtest)
  set_tests_properties(secant${test_arg}_mtest
    PROPERTIES DEPENDS MFrontCastemBehaviours)
  set_tests_properties(secant${test_arg}_mtest
    PROPERTIES DEPENDS mtest)
  add_custom_target(secant${test_arg}.mtest-install
    ${CMAKE_COMMAND} -E copy secant${test_arg}.mtest
    ${CMAKE_INSTALL_PREFIX}/share/mfront/tests/behaviours/secant/secant${test_arg}.mtest)
  add_dependencies(tests-install
    secant${test_arg}.mtest-install)
endmacro(test_secant)
